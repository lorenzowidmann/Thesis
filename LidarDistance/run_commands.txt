# --- Ubuntu: bring the Livox up, then measure distance ---

# 1. Set the NIC on the Livox subnet (interface name may differ; here enp89s0)
sudo ifconfig enp89s0 192.168.1.50

# 2. Configure/start the device once with LivoxViewer2 (sets host IP = 192.168.1.50),
#    confirm you see points, then CLOSE LivoxViewer2 (it holds the UDP port).
./LivoxViewer2.sh

# 3. Run the distance measurement (Python 3.10+, numpy)
cd Thesis/LidarDistance
python3 main.py                       # one 0.5 s measurement, 20 deg square (40*0.5)
python3 main.py --continuous          # keep measuring until Ctrl-C
python3 main.py --square-deg 20 --json # explicit 20 deg square, JSON output
python3 main.py --host-ip 192.168.1.50 --min-range 0.2 --max-range 30

# Notes:
# - Default data port is 57000 (Livox SDK2). Override with --data-port.
# - Distance reported = MEDIAN euclidean range of the returns inside the square.
# - Sensor frame: x forward (optical axis), y left, z up.
